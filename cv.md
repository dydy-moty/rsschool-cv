function spinWords(string $str): string {
     $arr = explode(' ',$str);
         foreach ($arr as $key=>$elem) {
             (strlen($elem) < 5) ? $arr[$key] = $elem : $arr[$key] = strrev($elem);
             }
     return $newStr = implode($arr, ' ');
 }
